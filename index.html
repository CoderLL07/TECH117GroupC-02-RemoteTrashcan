Introduction

This project uses an Arduino Uno, an HC-SR04 ultrasonic sensor, and servo motor to create a touchless trashcan, which prevents potential contamination of the can and allows for easy disposal.

Circuit Diagram

The ultrasonic sensor measures distance; the Arduino activates servo motor according readings.

Environmental Impact Assessment

  • Materials: The Arduino Uno is RoHS and WEEE compliant, limiting hazardous substances.
  • Energy Efficiency: The circuit draws from a 12v battery.
  • Reusability: All components are modular and reusable, reducing electronic waste.
  • Packaging & Shipping: taking from class parts kit.

Arduino Code

The following Arduino code controls the system, activating the motor by reading inputs from the sensor.

  
  
// Collision Warning System with Distance Sensor (HC-SR04)
// Author: Gerardo
// Oct 2, 2025
// Object is farther than 50cm: Green LED lights up
// Object is closer than 50 cm, but more than 10cm away: Yellow LED lights, buzzer plays 300Hz tone
// Object is closer than 10cm: Red LED lights up, buzzer plays 500Hz tone




#include 

// ----- Pins -----

const int TrigPin = 9;

const int EchoPin = 10;

const int servoPin = 6;

// ----- Objects -----

Servo lidServo;

// ----- Variables -----

float distanceCM = 0;

void setup() {

pinMode(TrigPin, OUTPUT);

pinMode(EchoPin, INPUT);

lidServo.attach(servoPin);

lidServo.write(0); // Lid starts closed

Serial.begin(9600);

}

void loop() {

// ----- Read Distance -----

digitalWrite(TrigPin, LOW);

delayMicroseconds(2);

digitalWrite(TrigPin, HIGH);

delayMicroseconds(10);

digitalWrite(TrigPin, LOW);

distanceCM = pulseIn(EchoPin, HIGH) / 58.0;

distanceCM = ((int)(distanceCM * 100.0)) / 100.0;

Serial.print("Distance = ");

Serial.print(distanceCM);

Serial.println(" cm");

// ----- Control Logic -----

if (distanceCM > 0 && distanceCM <= 30) {

Serial.println("Object detected — FAST opening.");

// ---- FAST OPEN ----

lidServo.write(90); // Snap open quickly

delay(300); // Fast open speed

// Stay open for 5 seconds

delay(5000);

Serial.println("Slow closing...");

// ---- SLOW CLOSE ----

for (int pos = 90; pos >= 0; pos--) {

lidServo.write(pos);

delay(25); // Slow animation (long close)

}

Serial.println("Lid closed.");

}

delay(500);

}
